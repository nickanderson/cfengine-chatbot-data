#+title: Example illustrating Inventory CFEngine Openssl Ciphers

#+caption: Example Policy
#+begin_src cfengine3 :include-stdlib t :log-level info :exports both :extra-opts --show-evaluated-vars=default:cfengine_openssl_ciphers_inventory\\.
  body file control
  {
        inputs => { "$(sys.libdir)/stdlib.cf" };
  }
  bundle agent __main__
  {
    methods:
        "cfengine_openssl_ciphers_autorun";
  }

  bundle agent cfengine_openssl_ciphers_autorun
  {
    meta:
      services_autorun|services_autorun_bundles::
        "tags" slist => { "autorun" };

    methods:
        "cfengine_openssl_ciphers_state";
        "cfengine_openssl_ciphers_inventory";

  }
  bundle agent cfengine_openssl_ciphers_inventory
  {
    vars:

      "l_cfengine_bin_openssl_ciphers"
        slist => readstringlist("$($(this.namespace):cfengine_openssl_ciphers_state.statefile)", "", "\n", inf, inf),
        if => fileexists( "$($(this.namespace):cfengine_openssl_ciphers_state.statefile)" );

      "arr_cfengine_bin_openssl_cipher[$(l_cfengine_bin_openssl_ciphers)]"
        string => "$(l_cfengine_bin_openssl_ciphers)",
        meta => { "inventory", "attribute_name=CFEngine Openssl Available Ciphers" },
        comment => "When restricting ciphers used for CFEngine communications it's useful to know what ciphers each host supports"; # Note, this is targeting the Openssl CFEngine uses
  }
  bundle agent cfengine_openssl_ciphers_state
  {
    vars:

        # /var/cfengine/bin/openssl ciphers
        "s_bin_openssl"
          string => ifelse( isexecutable( "$(sys.bindir)/openssl" ), "$(sys.bindir)/openssl",
                            #"_stdlib_path_exists_true", $(default:paths.openssl), # This does not work (special handling with skipping and ifelse() and 3 arguments only for vars that get skipped).
                            "/bin/openssl");

        "statefile" string => "$(sys.statedir)/cfengine_openssl_ciphers.dat";

    commands:
        "$(s_bin_openssl)"
          arglist => {
                       "ciphers",
                       "-v",
                       ">",
                       "$(statefile)",
          },
        contain => default:in_shell;
  }
#+end_src
